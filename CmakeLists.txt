cmake_minimum_required(VERSION 3.15)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_BUILD_TYPE Debug)

project(main)

set(Classes src/classes)
set(Handlers src/handlers)

add_executable(
    ${PROJECT_NAME} src/DataManagement.cpp src/Menu.cpp ${Handlers}/LoginHandler.cpp
    ${Handlers}/MainHandler.cpp ${Classes}/Person.cpp ${Classes}/Student.cpp 
    ${Classes}/Faculty.cpp)
target_include_directories(${PROJECT_NAME} PUBLIC /include)

find_library(mongocxx mongocxx.3.6.7 PATHS /usr/local/lib)
find_library(bsoncxx bsoncxx.3.6.7 PATHS /usr/local/lib)
find_library(ncurses ncurses PATHS /usr/local/lib)
target_include_directories(${PROJECT_NAME} PUBLIC
                           /usr/local/include/bsoncxx/v_noabi 
                           /usr/local/include/mongocxx/v_noabi
                           /usr/local/include)
target_link_libraries(${PROJECT_NAME} PUBLIC ${mongocxx} ${bsoncxx} ${ncurses})
